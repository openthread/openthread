79 PackageType_t
80 ChipInfo_GetPackageType( void )
81 {
82  PackageType_t packType = (PackageType_t)((
83  HWREG( FCFG1_BASE + FCFG1_O_USER_ID ) &
84  FCFG1_USER_ID_PKG_M ) >>
85  FCFG1_USER_ID_PKG_S ) ;
86 
87  if (( packType < PACKAGE_4x4 ) ||
88  ( packType > PACKAGE_WCSP ) )
89  {
90  packType = PACKAGE_Unknown;
91  }
92 
93  return ( packType );

96 //*****************************************************************************
97 //
98 // ChipInfo_GetChipFamily()
99 //
100 //*****************************************************************************
101 ChipFamily_t
102 ChipInfo_GetChipFamily( void )
103 {
104  ChipFamily_t chipFam = FAMILY_Unknown ;
105  uint32_t waferId ;
106 
107  waferId = (( HWREG( FCFG1_BASE + FCFG1_O_ICEPICK_DEVICE_ID ) &
108  FCFG1_ICEPICK_DEVICE_ID_WAFER_ID_M ) >>
109  FCFG1_ICEPICK_DEVICE_ID_WAFER_ID_S ) ;
110 
111  if ( waferId == 0xB99A ) {
112  if ( ChipInfo_GetDeviceIdHwRevCode() == 0xB ) {
113  chipFam = FAMILY_CC26xx_R2 ;
114  } else {
115  chipFam = FAMILY_CC26xx ;
116  }
117  } else if ( waferId == 0xB9BE ) chipFam = FAMILY_CC13xx ;
118  else if ( waferId == 0xBB41 ) chipFam = FAMILY_CC26xx_Aga ;
119  else if ( waferId == 0xBB20 ) chipFam = FAMILY_CC26xx_Liz ;
120 
121  return ( chipFam );
122 }

125 //*****************************************************************************
126 //
127 // ChipInfo_GetChipType()
128 //
129 //*****************************************************************************
130 ChipType_t
131 ChipInfo_GetChipType( void )
132 {
133  ChipType_t chipType = CHIP_TYPE_Unknown ;
134  uint32_t fcfg1UserId = ChipInfo_GetUserId();
135  uint32_t fcfg1Protocol = (( fcfg1UserId & FCFG1_USER_ID_PROTOCOL_M ) >>
136  FCFG1_USER_ID_PROTOCOL_S ) ;
137 
138  switch( ChipInfo_GetChipFamily() ) {
139 
140  case FAMILY_CC26xx :
141  switch ( fcfg1Protocol ) {
142  case 0x2 :
143  chipType = CHIP_TYPE_CC2620 ;
144  break;
145  case 0x4 :
146  chipType = CHIP_TYPE_CC2630 ;
147  case 0x1 :
148  case 0x9 :
149  chipType = CHIP_TYPE_CC2640 ;
150  if ( fcfg1UserId & ( 1 << 23 )) {
151  chipType = CHIP_TYPE_CUSTOM_1;
152  }
153  break;
154  case 0xF :
155  chipType = CHIP_TYPE_CC2650 ;
156  if ( fcfg1UserId & ( 1 << 24 )) {
157  chipType = CHIP_TYPE_CUSTOM_0;
158  }
159  break;
160  }
161  break;
162 
163  default :
164  chipType = CHIP_TYPE_Unknown ;
165  break;
166  }
167 
168  return ( chipType );
169 }

172 //*****************************************************************************
173 //
174 // ChipInfo_GetHwRevision()
175 //
176 //*****************************************************************************
177 HwRevision_t
178 ChipInfo_GetHwRevision( void )
179 {
180  HwRevision_t hwRev = HWREV_Unknown ;
181  uint32_t fcfg1Rev = ChipInfo_GetDeviceIdHwRevCode() ;
182  uint32_t minorHwRev = ChipInfo_GetMinorHwRev() ;
183 
184  switch( ChipInfo_GetChipFamily() ) {
185  case FAMILY_CC26xx :
186  switch ( fcfg1Rev ) {
187  case 1 : // CC26xx PG1.0
188  hwRev = HWREV_1_0;
189  break;
190  case 3 : // CC26xx PG2.0
191  hwRev = HWREV_2_0;
192  break;
193  case 7 : // CC26xx PG2.1
194  hwRev = HWREV_2_1;
195  break;
196  case 8 : // CC26xx PG2.2 (or later)
197  hwRev = (HwRevision_t)(((uint32_t)HWREV_2_2 ) + minorHwRev );
198  break;
199  }
200  break;
201  case FAMILY_CC13xx :
202  switch ( fcfg1Rev ) {
203  case 0 : // CC13xx PG1.0
204  hwRev = HWREV_1_0;
205  break;
206  case 2 : // CC13xx PG2.0 (or later)
207  hwRev = (HwRevision_t)(((uint32_t)HWREV_2_0 ) + minorHwRev );
208  break;
209  }
210  break;
211  case FAMILY_CC26xx_Liz :
212  case FAMILY_CC26xx_Aga :
213  switch ( fcfg1Rev ) {
214  case 0 : // CC26xx_Liz or CC26xx_Aga PG1.0 (or later)
215  hwRev = (HwRevision_t)(((uint32_t)HWREV_1_0 ) + minorHwRev );
216  break;
217  }
218  break;
219  case FAMILY_CC26xx_R2 :
220  hwRev = (HwRevision_t)(((uint32_t)HWREV_1_0 ) + minorHwRev );
221  break;
222  default :
223  // GCC gives warning if not handling all options explicitly in a "switch" on a variable of type "enum"
224  break;
225  }
226 
227  return ( hwRev );
228 }

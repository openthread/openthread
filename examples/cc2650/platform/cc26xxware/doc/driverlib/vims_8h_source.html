299 VIMSLineBufDisable(uint32_t ui32Base)
300 {
301  //
302  // Disable line buffers
303  //
304  HWREG(ui32Base + VIMS_O_CTL) |= VIMS_CTL_IDCODE_LB_DIS_M |
305  VIMS_CTL_SYSBUS_LB_DIS_M;
306 }
